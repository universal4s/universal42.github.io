基本原理

素数，又叫质数，除了1和它本身以外不再有其他的因数。

代码模板

1） 时间复杂度是O(n)

bool prime(int x){//判断x是不是质数，是返回true，不是返回false

if(x <= 1) return false;

for(int i = 2; i < x; i ++){

    if(x % i == 0) return false;

}

return true;

}

2）时间复杂度O( √n)

bool prime(int x){//判断x是不是质数，是返回true，不是返回false

if(x <= 1) return false;

for(int i = 2; i <= sqrt(x + 0.5); i ++){//0.5是防止根号的精度误差

    if(x % i == 0) return false;

}

return true;

}

3）时间复杂度O( √n)

bool prime(int x){//判断x是不是质数，是返回true，不是返回false

if(x <= 1) return false;

for(int i = 2; i * i <= x; i ++){//用乘法避免根号的精度误差

    if(x % i == 0) return false;

}

return true;

}

预处理每个数的质因数分解

#include

#include

using namespace std;

const int N = 100000 + 5;

vector prime_factor[N];

void init(){

int temp;

for(int i = 2; i < N; i ++){

    if(prime_factor\[i\].size() == 0){

        for(int j = i; j < N; j += i){

            temp = j;

            while(temp % i == 0){

                prime\_factor\[j\].push\_back(i);

                temp /= i;

            }  

        }

    }

}

}

int main(){

init();

}
